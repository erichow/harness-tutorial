import { Transcript } from "../harness/transcript.js";
import { ToolRegistry } from "../harness/tools/registry.js";
import { ToolCatalog } from "../harness/tools/catalog.js";
import { ContextAccountant } from "../harness/context/accountant.js";
import { Compactor } from "../harness/context/compactor.js";
import { PermissionManager } from "../harness/permissions/manager.js";
import { BudgetEnforcer } from "../harness/budget/enforcer.js";
import { Checkpointer } from "../harness/checkpoint/checkpointer.js";
import { createProvider } from "./provider.js";
import { cliPermissionHandler } from "./permissions.js";
import { cliEventHandler } from "./events.js";
import { readline } from "node:readline/promises";

export async function runAgentCLI(): Promise<void> {
  // 1. 注册所有基础工具
  const registry = new ToolRegistry();
  const catalog = new ToolCatalog(registry);
  registry.registerFileTools();
  registry.registerWebTools();
  registry.registerMCPTools();

  // 2. 权限管理器（交互式确认）
  const permissionManager = new PermissionManager();
  permissionManager.setHandler(cliPermissionHandler);

  // 3. 上下文管理
  const accountant = new ContextAccountant({ maxContextSize: 100_000 });

  // 4. Provider（从环境变量读取配置）
  const provider = createProvider();

  // 5. 预算
  const budget = new BudgetEnforcer({ enabled: true, maxTokens: 100_000 });

  // 6. 主交互循环
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
  });

  let transcript = new Transcript();
  try {
    console.log("Agent 已就绪。输入你的问题（输入 exit 退出）");

    while (true) {
      const input = await rl.question("> ");
      if (input.trim().toLowerCase() === "exit") break;

      transcript = await arun(
        provider,
        registry,
        input,
        transcript,
        "你是一个 coding agent。使用可用工具帮助用户完成编程任务。",
        cliEventHandler,
        undefined,  // onToolCall
        undefined,  // onToolResult
        undefined,  // onSnapshot
        accountant,
        undefined,  // compactor
        undefined,  // onCompaction
        catalog.getPinnedTools(),
        10,         // toolsPerTurn
      );
    }
  } finally {
    rl.close();
  }
}`}</CodeBlock>

import { readline } from "node:readline/promises";

export async function cliPermissionHandler(
  req: PermissionRequest,
): Promise<PermissionDecision> {
  // 自动允许读操作
  if (req.action.startsWith("read_") || req.action === "list_directory") {
    return { allowed: true, reason: "read-only" };
  }

  console.log(\`\n⚠️  需要确认: \${req.action}\`);
  if (req.details) {
    console.log(\`   \${req.details}\`);
  }

  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
  });

  try {
    const answer = await rl.question("  允许? (y=一次 / a=总是允许 / n=拒绝 / v=查看详情): ");

    switch (answer.trim().toLowerCase()) {
      case "y": return { allowed: true, remember: false };
      case "a": return { allowed: true, remember: true };
      case "v":
        console.log(JSON.stringify(req, null, 2));
        const followUp = await rl.question("  现在允许? (y/n): ");
        return { allowed: followUp.trim().toLowerCase() === "y", remember: false };
      default:
        return { allowed: false, reason: "用户拒绝" };
    }
  } finally {
    rl.close();
  }
}`}</CodeBlock>

export function cliEventHandler(event: StreamEvent): void {
  switch (event.type) {
    case "text":
      process.stdout.write(event.delta);
      break;

    case "tool_call":
      console.log(\`\n🔧 正在使用工具: \${event.toolName}\`);
      break;

    case "tool_result":
      const preview = event.content.slice(0, 200);
      console.log(\`📎 工具返回: \${preview}\${event.content.length > 200 ? "…" : ""}\`);
      break;

    case "error":
      console.error(\`❌ 错误: \${event.message}\`);
      break;

    case "progress":
      // 第 29 章会实现进度条
      break;

    case "compaction":
      console.log(\`📦 压缩: 回收了 \${event.tokensReclaimed} tokens\`);
      break;
  }
}`}</CodeBlock>

        <Callout kind="tip" title="流式渲染是最好的 UX">
