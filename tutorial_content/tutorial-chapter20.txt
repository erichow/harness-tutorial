@dataclass
class ModelRouter:
    economy: Provider
    mid: Provider
    premium: Provider

    def choose(self, transcript: Transcript, task_hint: str | None = None) -> Provider:

      <Section id="ch20-3" num="20.3" title="BudgetEnforcer（三件里最难、不可缺）">
        <p>三件里最难的，<strong>少了它另外两件都是装饰</strong>。一个跑飞的 loop 在<em>内层</em>生成成本，不是 turn 边界——在 turn 起点的 enforcement 检查不能停掉正在跑的 turn。</p>

        <Callout kind="key" title="能 work 的模式：在独立任务里强制">
          <p>主 loop 跑 agent。一个 watchdog 跟踪 session 成本；上限被达到时，它 cancel 主任务。</p>

        <CodeBlock filename="src/harness/cost/enforcer.py" lang="python">{`class BudgetExceeded(Exception):
    pass


@dataclass
class BudgetEnforcer:
    max_usd: float                    # 硬性 session 上限
    alert_thresholds: list[float] = field(default_factory=lambda: [0.5, 0.8])
    spent_usd: float = field(default=0.0, init=False)
    alerted: set[float] = field(default_factory=set, init=False)
    _cancelled_task: "asyncio.Task | None" = field(default=None, init=False)

    def attach_task(self, task: asyncio.Task) -> None:
        self._cancelled_task = task

    def record(self, input_tokens: int, output_tokens: int, model: str) -> None:
        cost = self._price(model, input_tokens, output_tokens)
        self.spent_usd += cost

        for t in self.alert_thresholds:
            if t not in self.alerted and self.spent_usd / self.max_usd >= t:
                self.alerted.add(t)
                print(f"[BUDGET WARNING] {self.spent_usd:.2f} / {self.max_usd:.2f} "
                      f"({t*100:.0f}% reached)")

        if self.spent_usd >= self.max_usd:
            self._halt()

    def _halt(self) -> None:
        if self._cancelled_task and not self._cancelled_task.done():
            print(f"[BUDGET HALT] {self.spent_usd:.2f} exceeds {self.max_usd:.2f}; cancelling session")
            self._cancelled_task.cancel()
        raise BudgetExceeded(
            f"session budget budget {self.max_usd} exceeded: {self.spent_usd:.2f}"
        )

    def _price(self, model: str, in_toks: int, out_toks: int) -> float:
        # 2026.4 价格。把它移到带日期的 appendix 或配置文件，
        # 不要让它在 harness 代码里腐烂。
        prices = {
            "claude-sonnet-4-6": (3.0,  15.0),
            "claude-opus-4-6":   (5.0,  25.0),
            "claude-haiku":      (0.8,   4.0),
            "gpt-5":             (1.25, 10.0),
            "gpt-5.2":           (1.75, 14.0),
            # local / free providers —— 零价让 Ch 7-17 的 LocalProvider demo
            # 不要日志成虚假的 Opus 价
            "local":             (0.0,  0.0),
            "stub":              (0.0,  0.0),
        }
        # 未知模型 fallback：Opus tier，刻意的。**未知 provider 高报成本好过低报**。
        in_rate, out_rate = prices.get(model, (5.0, 25.0))
        return (in_toks * in_rate + out_toks * out_rate) / 1_000_000`}</CodeBlock>

        <Callout kind="key" title="为什么同时 cancel() 和 raise">
          <p><strong>raise</strong>：<code>record()</code> 同步跑在 loop 自己的栈里——<code>raise BudgetExceeded</code> 是真正停掉本 session 的东西：它沿 <code>record()</code> → arun loop → caller 的 <code>await arun(...)</code> 上传。</p>
          <p><strong>cancel()</strong>：是为<em>另一个场景</em>留的——当 parent coroutine 在 <code>await asyncio.gather(...)</code> 多个 agent 时（§17.4 的并行 spawner）。<code>raise</code> 停掉这条栈，但<strong>sibling session 跑在其它任务上会继续烧 token</strong>直到它们自己的 turn 结束。<code>cancel()</code> 把 <code>asyncio.CancelledError</code> 传播到 sibling，让它们也停。</p>
          <p>Caller 把 <code>await arun(...)</code> 包在 try/except 里同时抓 <code>BudgetExceeded</code>（本 session 撞 cap）和 <code>asyncio.CancelledError</code>（sibling 撞 cap 把我们也带下来）。</p>
        </Callout>

        <Callout kind="warn" title="未知模型 fallback 的副作用">
          <p><code>prices.get(model, (5.0, 25.0))</code> 把不在表里的全当 Opus tier。这是<strong>刻意的安全选择</strong>——对未知 provider 低报成本比高报危险——但意味着<em>adapter 名字不在表里时（一个用户加的 <code>GroqProvider.name == "groq"</code>）会日志成虚高成本，直到你加一行</em>。</p>
          <p><strong><code>provider.name = "something"</code> 出现 bewildering bill 时，第一个嫌疑人是表里缺行，不是真的 overrun</strong>。<code>"local"</code> 和 <code>"stub"</code> 预设为零正是为了让本书自己的 LocalProvider demo（Ch 7、8、12、17）不产生误导成本数字。</p>
        </Callout>

        <CodeBlock filename="src/harness/agent.py（budget-aware sketch）" lang="python">{`async def arun(..., budget_enforcer: "BudgetEnforcer | None" = None) -> str:
    current = asyncio.current_task()
    if budget_enforcer and current:
        budget_enforcer.attach_task(current)

    # ... 既有 setup
    for _ in range(MAX_ITERATIONS):
        # ... turn 执行
        response = await _one_turn(...)
        if budget_enforcer:
            budget_enforcer.record(
                input_tokens=response.input_tokens,
                output_tokens=response.output_tokens,
                model=provider.name,
            )
        # ... continue`}</CodeBlock>

        <Callout kind="note">
          一个微妙处：<code>record()</code> 在每 turn 之后同步触发。一个本身 10 秒、产 100K output token 的 turn <em>在 record 跑之前就已经贵了</em>。对多数情况没问题——下个 turn 被 halt。对一个 turn 单独就破预算的病理情况，你要加流式 enforce 看 output token 到达时的状态。<em>本书不建；turn 边界硬 cap 接住 95% 的真实 runaway 模式</em>。
        </Callout>
