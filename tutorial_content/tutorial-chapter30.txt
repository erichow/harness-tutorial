  colorEnabled: true,

/* ──────────── 加载器 ──────────── */

export async function loadConfig(): Promise<AgentConfig> {
  const configs: Partial<AgentConfig>[] = [DEFAULTS];

  // 1. 全局配置
  const globalConfig = await tryReadJSON(
    join(homedir(), ".agent-code", "config.json"),
  );
  if (globalConfig) configs.push(globalConfig);

  // 2. 项目配置
  const projectConfig = await tryReadJSON(
    resolve(process.cwd(), ".agent-code.json"),
  );
  if (projectConfig) configs.push(projectConfig);

  const projectDirConfig = await tryReadJSON(
    resolve(process.cwd(), ".agent-code", "config.json"),
  );
  if (projectDirConfig) configs.push(projectDirConfig);

  // 3. 环境变量
  configs.push(loadEnvOverrides());

  // 4. CLI 参数
  configs.push(loadCLIOverrides());

  // 合并
  return mergeConfigs(configs);
}

/* ──────────── 辅助函数 ──────────── */

async function tryReadJSON(path: string): Promise<Partial<AgentConfig> | null> {
  try {
    await access(path);
    const content = await readFile(path, "utf-8");
    return JSON.parse(content);
  } catch {
    return null;
  }
}

function loadEnvOverrides(): Partial<AgentConfig> {
  const env = process.env;
  const overrides: Partial<AgentConfig> = {};

  if (env.PROVIDER) overrides.provider = env.PROVIDER;
  if (env.MODEL) overrides.model = env.MODEL;
  if (env.TEMPERATURE) overrides.temperature = parseFloat(env.TEMPERATURE);
  if (env.MAX_TOKENS_PER_TURN) overrides.maxTokensPerTurn = parseInt(env.MAX_TOKENS_PER_TURN);
  if (env.TOOLS) overrides.enabledTools = env.TOOLS.split(",");
  if (env.BUDGET_ENABLED === "true") overrides.budgetEnabled = true;
  if (env.THINKING_MODE) overrides.thinkingMode = env.THINKING_MODE as any;

  return overrides;
}

function loadCLIOverrides(): Partial<AgentConfig> {
  try {
    const args = parseArgs({
      args: process.argv.slice(2),
      options: {
        provider: { type: "string" },
        model: { type: "string" },
        temperature: { type: "string" },
        maxTokensPerTurn: { type: "string" },
        "thinking-mode": { type: "string" },
        "no-color": { type: "boolean" },
      },
      strict: false,
    });

    const overrides: Partial<AgentConfig> = {};
    if (args.values.provider) overrides.provider = args.values.provider as string;
    if (args.values.model) overrides.model = args.values.model as string;
    if (args.values.temperature) overrides.temperature = parseFloat(args.values.temperature as string);
    if (args.values["thinking-mode"]) overrides.thinkingMode = args.values["thinking-mode"] as any;
    if (args.values["no-color"]) overrides.colorEnabled = false;

    return overrides;
  } catch {
    return {};
  }
}

function mergeConfigs(layers: Partial<AgentConfig>[]): AgentConfig {
  const result = { ...DEFAULTS };
  for (const layer of layers) {
    for (const [key, value] of Object.entries(layer)) {
      if (value !== undefined) {
        (result as any)[key] = value;
      }
    }
  }
  return result;
}`}</CodeBlock>

export interface AgentConfig {
  // ...前面已有的...

  // 权限策略
  readCommands: string[];   // 自动放行的命令前缀
  writeCommands: string[];  // 需确认的命令前缀
  blockedCommands: string[]; // 禁止执行的命令
}

const DEFAULTS: AgentConfig = {
  // ...
  readCommands: [
    "ls", "cat", "head", "tail", "echo", "pwd", "which",
    "type", "node --version", "npm --version", "git --version",
  ],
  writeCommands: [
    "rm", "mv", "cp", "chmod", "chown", "mkdir", "touch",
    "git add", "git commit", "git push", "git stash",
    "npm install", "pip install",
  ],
  blockedCommands: [
    "sudo", "su", "chmod 777", "curl.*| sh", "wget.*| bash",
  ],
};

// 在 permission handler 中使用
export function classifyCommand(
  command: string,
  config: AgentConfig,
): "read" | "write" | "blocked" | "unknown" {
  const trimmed = command.trimStart();

  for (const pattern of config.blockedCommands) {
    if (new RegExp(pattern).test(trimmed)) return "blocked";
  }
  for (const readCmd of config.readCommands) {
    if (trimmed.startsWith(readCmd)) return "read";
  }
  for (const writeCmd of config.writeCommands) {
    if (trimmed.startsWith(writeCmd)) return "write";
  }
  return "unknown";
}`}</CodeBlock>

        <Callout kind="warn" title="blockedCommands 不是安全边界">
          <p>命令黑名单（blockedCommands）很容易绕过——用户可以用 base64 编码、shell 脚本、<code>/full/path/to/sudo</code> 等方法跳过字符串匹配。<strong>真正的安全来自 PermissionManager 的交互式确认 + 最低权限原则</strong>，而不是命令名匹配。blockedCommands 只是一个给用户的"快捷拒绝清单"。</p>
