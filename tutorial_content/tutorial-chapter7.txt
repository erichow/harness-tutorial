function ContextLiveDemo() {

  const [schemas, setSchemas] = useS7(2000);

  const u = total / usable;
  const state = u >= 0.8 ? "red" : u >= 0.6 ? "yellow" : "green";
  const stateColor = { green: "var(--green)", yellow: "var(--accent-3)", red: "var(--red)" }[state];

import json
from dataclasses import dataclass, field
from typing import Literal

import tiktoken

from ..messages import (
    Block, Message, ReasoningBlock, TextBlock,
    ToolCall, ToolResult, Transcript,
)

Component = Literal["system", "tools", "history", "retrieved", "headroom"]


@dataclass
class ContextBudget:
    window_size: int = 200_000
    headroom: int = 4096          # reserved for the model's response
    yellow_threshold: float = 0.60
    red_threshold: float = 0.80

    @property
    def usable(self) -> int:
        return self.window_size - self.headroom


@dataclass
class ContextSnapshot:
    totals: dict[Component, int] = field(default_factory=dict)
    budget: ContextBudget = field(default_factory=ContextBudget)

    @property
    def total_used(self) -> int:
        return sum(v for k, v in self.totals.items() if k != "headroom")

    @property
    def utilization(self) -> float:
        return self.total_used / max(self.budget.usable, 1)

    @property
    def state(self) -> Literal["green", "yellow", "red"]:
        u = self.utilization
        if u >= self.budget.red_threshold:    return "red"
        if u >= self.budget.yellow_threshold: return "yellow"
        return "green"


class ContextAccountant:
    """Counts tokens per component across a transcript."""

    def __init__(self, encoding_name: str = "cl100k_base",
                 budget: ContextBudget | None = None) -> None:
        self._enc = tiktoken.get_encoding(encoding_name)
        self.budget = budget or ContextBudget()

    def snapshot(self, transcript: Transcript,
                 tools: list[dict] | None = None,
                 retrieved: list[str] | None = None) -> ContextSnapshot:
        totals: dict[Component, int] = {
            "system":    self._count_text(transcript.system or ""),
            "tools":     sum(self._count_text(json.dumps(t)) for t in (tools or [])),
            "history":   sum(self._count_message(m) for m in transcript.messages),
            "retrieved": sum(self._count_text(r) for r in (retrieved or [])),
            "headroom":  self.budget.headroom,
        }
        return ContextSnapshot(totals=totals, budget=self.budget)

    def _count_text(self, s: str) -> int:
        return len(self._enc.encode(s))

    def _count_message(self, m: Message) -> int:
        total = 4   # ~4 tokens of per-message overhead in most provider formats
        for block in m.blocks:
            total += self._count_block(block)
        return total

    def _count_block(self, block: Block) -> int:
        match block:
            case TextBlock(text=t):        return self._count_text(t)
            case ToolCall(name=n, args=a):
                return self._count_text(n) + self._count_text(json.dumps(a)) + 6
            case ToolResult(content=c):    return self._count_text(c) + 4
            case ReasoningBlock(text=t):
                # Counted when an adapter preserves reasoning on the transcript
                # (Anthropic thinking + tools, or explicit consumer choice).
                return self._count_text(t)
            case _:
                # Defensive: new block types added later should be undercounted,
                # not crash measurement.
                return 0`}</CodeBlock>

          跑起来，你会看到使用率回合接回合地涨。跳变来自<strong>工具输出</strong>——一份 <code>/proc/cpuinfo</code> 大约 5000 tokens，一次调用就让进度条挪动几个百分点。用一个读三个大文件的 prompt 重复，你能<em>实时看着它走向 yellow</em>。这就是要点：本来看不见的，现在你能盯着了。
        </Callout>
      </Section>

      <Section id="ch7-7" num="7.7" title="值得记的观察">
        <div className="cards cols-3">
          <Card title="工具输出占大头" kind="">
            多数 agent 工作负载里，到回合 10，<strong>tool 结果是 transcript 的 70-90%</strong>。System / schemas 都是误差。所以第 11 章一整章给"工具输出设计"——更小、更结构化的输出是<em>上下文里 ROI 最高的干预</em>。
          </Card>
          <Card title="用户消息小到可以忽略" kind="">
            人类每回合写一段文字，模型读几 KB。这种不对称解释了为什么"让窗口更大"的朴素直觉失败——<strong>填窗口的不是用户</strong>。
          </Card>
          <Card title="Assistant reasoning 是第 3 个鼓包" kind="">
            当你跑会"想出声"的 agent（extended thinking、verbose ReAct），<strong>assistant 自己的文字能逼近工具结果的大小</strong>。"记不记 reasoning"是开始记账之后才能做的有意识决定。
          </Card>
        </div>
      </Section>

      <Section id="ch7-8" num="7.8" title="prompt cache 折扣呢？">
        <Callout kind="note">
          <p>Anthropic 和 OpenAI 都支持 prompt caching：长且稳定的前缀（system + tool schemas）可以被 cache，后续共享前缀的请求按 ~10% 的 input rate 计费。</p>
          <p><strong>但 cached 前缀仍占同样的窗口空间</strong>——caching 是计费优化，不是窗口优化。Accountant 不论 cache 状态都数原始 token。要单独追踪 cache 有效成本？第 20 章引入 <code>CostAccountant</code>。</p>
        </Callout>
      </Section>


      <Section id="ch7-reconcile" num="7.6.b" title="实测对账：accountant 估算 vs provider 真实值">
        <p>看不见的偏差比看得见的偏差危险。tiktoken 的本地估算和 Anthropic / OpenAI 的真实 <code>input_tokens</code> 偶尔会差 3-8%。开发期跑一段对账是诚实工程：</p>
        <CodeBlock filename="examples/ch07_reconcile.py" lang="python">{`async def reconcile_run(prompt: str):
    accountant = ContextAccountant()
    rows = []

    def on_snapshot(snap):
        # snapshot 在每个 turn 开始时（调 provider 之前）触发
        rows.append({"estimated_in": snap.totals["history"] + snap.totals["system"]
                                     + snap.totals["tools"]})

    result = await arun(
        provider=AnthropicProvider(), catalog=ToolCatalog(tools=STANDARD_TOOLS),
        user_message=prompt, accountant=accountant, on_snapshot=on_snapshot,
    )

    # （从 Ch 18 的 OTel span 拉，或在 arun 里加一个 on_completion 回调）
    print(f"{'turn':>4} {'estimated':>10} {'actual':>10} {'delta_%':>8}")
    for i, (est, act) in enumerate(zip(rows, result.actual_input_per_turn)):
        delta = (act - est) / max(est, 1) * 100
        print(f"{i:>4} {est:>10,d} {act:>10,d} {delta:>+7.1f}%")`}</CodeBlock>
        <Callout kind="key" title="预期偏差范围">
          <ul style={{ margin: 0 }}>
            <li><strong>OpenAI 模型 + cl100k_base / o200k_base</strong>: ≤ 0.5%（字节精确）</li>
