from dataclasses import dataclass, field
from typing import Callable


@dataclass
class EvalCase:
    """A single golden trajectory test."""
    id: str
    description: str
    user_message: str
    system: str | None = None

    # Optional: 这些工具必须被调到（任意顺序）
    required_tools: list[str] = field(default_factory=list)

    # Optional: 这些工具必须不被调
    forbidden_tools: list[str] = field(default_factory=list)

    # Optional: 拿最终答案返回 True/False 的可调用——确定答案任务
    check_answer: Callable[[str], bool] | None = None

    # Optional: total tokens 上限。pass 如果在它之下
    max_tokens: int | None = None

    # Optional: iterations 上限
    max_iterations: int | None = None


@dataclass
class EvalResult:
    case_id: str
    passed: bool
    failures: list[str]
    final_answer: str
    tokens_used: int
    iterations_used: int
    duration_seconds: float`}</CodeBlock>

        <Callout kind="note">

        </Callout>

      <Section id="ch19-3" num="19.3" title="Runner">
        <CodeBlock filename="src/harness/evals/runner.py" lang="python">{`@dataclass
class EvalRunner:
    provider: Provider
    catalog: ToolCatalog

    async def run_one(self, case: EvalCase) -> EvalResult:
        start = time.time()
        tool_calls_observed: list[str] = []

        # Wrap catalog so dispatched tool names get appended to
        # tool_calls_observed. A ToolCatalog with observed-dispatch
        # is the simplest in-harness way to see what model called.
        recording_catalog = _RecordingCatalog(self.catalog, tool_calls_observed)

        try:
            result = await arun(
                provider=self.provider,
                catalog=recording_catalog,
                system=case.system,
                user_message=case.user_message,
            )
        except Exception as e:
            return EvalResult(
                case_id=case.id, passed=False,
                failures=[f"crashed: {type(e).__name__}: {e}"],
                final_answer="", tokens_used=0, iterations_used=0,
                duration_seconds=time.time() - start,
            )

        failures: list[str] = []
        if case.check_answer is not None and not case.check_answer(result.summary):
            failures.append("answer check failed")
        for required in case.required_tools:
            if required not in tool_calls_observed:
                failures.append(f"required tool not called: {required}")
        for forbidden in case.forbidden_tools:
            if forbidden in tool_calls_observed:
                failures.append(f"forbidden tool called: {forbidden}")
        if case.max_tokens is not None and result.tokens_used > case.max_tokens:
            failures.append(f"tokens_used {result.tokens_used} > {case.max_tokens}")
        if case.max_iterations is not None and result.iterations_used > case.max_iterations:
            failures.append(f"iterations_used {result.iterations_used} > {case.max_iterations}")

        return EvalResult(
            case_id=case.id,
            passed=len(failures) == 0,
            failures=failures,
            final_answer=result.summary,
            tokens_used=result.tokens_used,
            iterations_used=result.iterations_used,
            duration_seconds=time.time() - start,
        )

    async def run_all(self, cases: list[EvalCase]) -> list[EvalResult]:
        results = []
        for case in cases:
            result = await self.run_one(case)
            print(f"{'✓' if result.passed else '✗'} {case.id}: {case.description} "
                  f"[{result.tokens_used} tok, {result.duration_seconds:.1f}s]"
                  + (f" — {', '.join(result.failures)}" if result.failures else ""))
            results.append(result)
        return results`}</CodeBlock>

        <Callout kind="key" title="_RecordingCatalog —— 关键拦截点">
          <p>Catalog 接口是 <code>select(query, k, must_include)</code> 和 <code>get(name)</code>。<strong>包装 <code>select</code> 返回的工具</strong>是干净的拦截点：每个返回的 Tool 的 <code>arun</code>/<code>run</code> 被 wrap 来记录名字、再 delegate。</p>
          <p><em>如果用 §12.5 的 discovery 工具（它直接读 <code>catalog.tools</code>）或者从别处访问 <code>catalog.get(name)</code>/<code>catalog.all_names()</code>，也要 proxy 它们——每个都是一行代码</em>。</p>
        </Callout>

        <Callout kind="note">
          Tokens 和 tool-call 观测在 sketch 里是手挥。<strong>生产 eval runner 直接从 OTel span 拉</strong>——第 18 章建的 tracing 是对的底座。一个监听 <code>ConsoleSpanProcessor</code>-like collector 报 per-run metrics 的小 span-reader 约 50 行。
        </Callout>

      <Section id="ch19-5" num="19.5" title="LLM-as-Judge">
        <p>对 <code>check_answer</code> 是<em>主观</em>的任务——"总结这文章"——我们用另一个 LLM 当 judge。设计好 prompt + 比被测更强的模型：</p>

        <CodeBlock filename="src/harness/evals/llm_judge.py" lang="python">{`async def judge(
    judge_provider: Provider,
    question: str,
    candidate_answer: str,
    reference_answer: str | None = None,
    criteria: str = "accuracy, completeness, relevance",
) -> bool:
    transcript = Transcript(system=(
        "You are a strict evaluator. Given a question and a candidate answer, "
        "judge whether the answer is correct by the criteria provided. "
        "Reply with only 'PASS' or 'FAIL' followed by a one-sentence reason."
    ))
    user = (f"Question: {question}\\n\\n"
            f"Candidate answer: {candidate_answer}\\n\\n")
    if reference_answer:
        user += f"Reference answer for comparison: {reference_answer}\\n\\n"
    user += f"Criteria: {criteria}"
    transcript.append(Message.user_text(user))

    response = await judge_provider.acomplete(transcript, tools=[])
    text = response.text or ""
    return text.strip().upper().startswith("PASS")`}</CodeBlock>

        <Callout kind="warn" title="两个值得知道的 caveat">
          <p><strong>Judge bias</strong>：用 Claude judge Claude 的输出会让 judge 和 candidate 的错误相关——它们有共同盲点时，judge 漏掉失败。<em>最佳实践：judge 用不同 provider</em>（Claude judge GPT，或反过来）。</p>
          <p><strong>Judge ceiling</strong>：LLM judge 不能可靠超出它自己在底层任务上的能力天花板。<em>比 candidate 还弱的 judge 在难题上会自信地误评</em>。</p>
        </Callout>

        <Callout kind="note">
          本书场景里，确定性的 <code>check_answer</code> 函数 cover 多数情况。LLM-as-judge 是工具箱里的一件——<em>有函数能搞定时别伸手要它</em>。
        </Callout>
      </Section>

      <Section id="ch19-6" num="19.6" title="Production-to-Eval pipeline">
        <p>第 18 章的可观测性给我们结构化 trace 数据。一次<em>失败</em>的生产跑——崩了、超时、产出明显错答——是个潜在 eval case。一段小脚本把失败 trace 变成 <code>EvalCase</code>：</p>

        <CodeBlock filename="src/harness/evals/from_trace.py" lang="python">{`def case_from_trace(trace_summary: dict) -> EvalCase:
    """把生产 trace 转成回归 eval case。

    trace_summary: 从 tracing 后端抽出的 dict。典型字段：
                   user_message, system, final_answer, failure_reason.
    """
    return EvalCase(
        id=f"prod-regression-{trace_summary['trace_id'][:8]}",
        description=f"regression from production: "
                    f"{trace_summary.get('failure_reason', 'unknown')}",
        user_message=trace_summary["user_message"],
        system=trace_summary.get("system"),
        max_tokens=int(trace_summary.get("tokens_used", 0) * 1.5),
        # check 常常就是"别再重犯同样失败"。更复杂的可以查具体已知坏行为。
    )`}</CodeBlock>

        <Callout kind="key">
          工作流：<em>监控 flag 一个失败 trace → 工程师 review → 确认是要防的回归 → 跑 <code>case_from_trace</code> → review 生成的 case → 微调 → commit 到 suite</em>。下次 CI 跑这个 case；<strong>未来同类回归在 ship 前被 CI gate 挡住</strong>。
          <br /><br />这就是 eval suite 怎么有机生长。<em>每一次生产失败都在 suite 里留下化石</em>。久而久之，suite 编码了你系统见过的、最可能复发的具体失败模式。
        </Callout>
      </Section>

        <CodeBlock filename="src/harness/evals/stability.py" lang="python">{`async def run_stability(runner: EvalRunner, case: EvalCase, n: int = 10) -> dict:
    """Run the same case n times; report pass rate and per-run details."""
    results = []
    for i in range(n):
        results.append(await runner.run_one(case))
    passed = sum(1 for r in results if r.passed)
    return {
        "case_id": case.id,
        "pass_rate": f"{passed}/{n}",
        "per_run": ["✓" if r.passed else "✗" for r in results],
        "avg_tokens": sum(r.tokens_used for r in results) / n,
        "avg_duration": sum(r.duration_seconds for r in results) / n,
    }`}</CodeBlock>
