from dataclasses import dataclass
from typing import Protocol


@dataclass(frozen=True)
class ProviderResponse:
    """What a provider gives us back: either text, or a tool call."""
    kind: str  # "text" or "tool_call"
    text: str | None = None
    tool_name: str | None = None
    tool_args: dict | None = None
    tool_call_id: str | None = None


class Provider(Protocol):
    def complete(
        self, transcript: list[dict], tools: list[dict]
    ) -> ProviderResponse:
        """Given a transcript and available tools, produce one response."""
        ...`}</CodeBlock>

          <p><strong>ProviderResponse 把两种情况塞进同一个 shape</strong>。真实响应有 token 计数、finish reason、reasoning trace——但 loop 此刻只想知道一件事：调工具，还是给答案？其他都是别人的问题。</p>

        <CodeBlock filename="src/harness/providers/mock.py" lang="python">{`from __future__ import annotations
from .base import Provider, ProviderResponse


class MockProvider(Provider):
    """A scripted provider for teaching and testing.
    Walks through a fixed list of responses, one per call.
    """

    def __init__(self, responses: list[ProviderResponse]) -> None:
        self._responses = list(responses)
        self._index = 0

    def complete(
        self, transcript: list[dict], tools: list[dict]
    ) -> ProviderResponse:
        if self._index >= len(self._responses):
            raise RuntimeError("mock ran out of responses")
        response = self._responses[self._index]
        self._index += 1
        return response`}</CodeBlock>

        <Callout kind="note" title="为什么 class MockProvider(Provider) 还要继承？">
          Python 的 Protocol 是<strong>结构性</strong>的——只要方法名匹配就算实现。继承的两个理由：①<strong>表明意图</strong>，读代码的人一眼看到"这是 Provider 实现"；②<strong>把不匹配从隐藏 bug 升级成 mypy/pyright 错误</strong>——少写一个方法或改了签名，立刻报错而不是等 loop 跑起来时炸。
        </Callout>

        <p>
          <strong>朴素</strong> 这个词是故意的——它马上就要以 5 种方式破。但它仍是有用的起点，因为接下来缺的每一样东西都有具体的失败动机。

        <CodeBlock filename="src/harness/agent.py" lang="python">{`from __future__ import annotations
from typing import Callable
from .providers.base import Provider, ProviderResponse

MAX_ITERATIONS = 20


def run(
    provider: Provider,
    tools: dict[str, Callable[..., str]],
    tool_schemas: list[dict],
    user_message: str,
) -> str:
    transcript: list[dict] = [{"role": "user", "content": user_message}]

    for _ in range(MAX_ITERATIONS):
        response = provider.complete(transcript, tool_schemas)

        if response.kind == "text":
            transcript.append({"role": "assistant", "content": response.text})
            return response.text or ""

        if response.kind == "tool_call":
            if response.tool_name is None:
                raise RuntimeError("tool_call response is missing tool_name")
            if response.tool_name not in tools:
                raise RuntimeError(f"unknown tool: {response.tool_name!r}")

            tool_fn = tools[response.tool_name]
            result = tool_fn(**(response.tool_args or {}))

            transcript.append({
                "role": "assistant",
                "content": [{"type": "tool_use",
                             "name": response.tool_name,
                             "id": response.tool_call_id,
                             "input": response.tool_args}]
            })
            transcript.append({
                "role": "user",
                "content": [{"type": "tool_result",
                             "tool_use_id": response.tool_call_id,
                             "content": result}]
            })
            continue

        raise RuntimeError(f"unexpected response kind: {response.kind!r}")

    raise RuntimeError(f"agent did not finish in {MAX_ITERATIONS} iterations")`}</CodeBlock>

        <Callout kind="warn" title="读这段 loop 两次——后面整本书都在解决它的隐含选择">
          <ul style={{ margin: 0 }}>
            <li>transcript 是 list of dicts——<strong>没类型、没校验、没块结构</strong></li>
            <li>tools 是 <code>dict[str, Callable]</code>——<strong>没 schema 检查、没副作用声明、没权限闸门</strong></li>
            <li>工具被 <code>**response.tool_args</code> 隐式信任</li>
            <li>result 是字符串，<strong>哪怕它有 50000 字符</strong>，也直接进 transcript</li>
            <li>没有 streaming、cancellation、retry、可观测性、成本统计</li>
            <li><code>MAX_ITERATIONS = 20</code> 是 loop 与"失控成本"之间唯一的栏杆</li>
          </ul>

from harness.providers.base import ProviderResponse
from harness.providers.mock import MockProvider


def calc(expression: str) -> str:
    # dangerous in real life; fine for a mock
    return str(eval(expression, {"__builtins__": {}}, {}))


mock = MockProvider([
    ProviderResponse(
        kind="tool_call",
        tool_name="calc",
        tool_args={"expression": "2 + 2"},
        tool_call_id="call-1",
    ),
    ProviderResponse(kind="text", text="2 + 2 is 4."),
])

tool_schemas = [{
    "name": "calc",
    "description": "Evaluate a Python arithmetic expression.",
    "input_schema": {
        "type": "object",
        "properties": {"expression": {"type": "string"}},
        "required": ["expression"],
    },
}]

answer = run(
    provider=mock,
    tools={"calc": calc},
    tool_schemas=tool_schemas,
    user_message="What is 2 + 2?",
)
print(answer)  # -> "2 + 2 is 4."`}</CodeBlock>

        <FiveBreaksDiagram />

        <div className="cards cols-2">
          <Card title="Break 1 · 模型叫了一个不存在的工具" kind="bad">
            <p><code>tool_name="calculator"</code>（不是 "calc"）</p>
            <p>→ <code>RuntimeError: unknown tool: 'calculator'</code></p>
            <p>错误清楚了，但模型<strong>没机会恢复</strong>——异常把整个 loop 拆掉，前面 9 个回合的有用工作没了。</p>
            <p><strong>缺：</strong>把"unknown tool"作为结构化错误返回给模型的派发层。<em>→ 第 4 章 ToolRegistry</em></p>
          </Card>

          <Card title="Break 2 · 工具参数 schema 不匹配" kind="bad">
            <p><code>tool_args=&#123;"expr": "2+2"&#125;</code>（key 错了）</p>
            <p>→ <code>TypeError: calc() got an unexpected keyword argument 'expr'</code></p>
            <p>跟 Break 1 同类，但更深一层——模型搞错了参数名，但<strong>看不到自己错在哪</strong>。</p>
            <p><strong>缺：</strong>调用前的 schema 校验，把 ValidationError 喂回模型。<em>→ 第 6 章</em></p>
          </Card>

          <Card title="Break 3 · 工具自己抛异常" kind="bad">
            <p><code>tool_args=&#123;"expression": "1 / 0"&#125;</code></p>
            <p>→ <code>ZeroDivisionError</code>，loop 死</p>
            <p>更隐蔽的版本：网络超时、文件权限、shell exit code 1、远程 503——<strong>所有"做点有意思的事"都会遇到</strong>，但 loop 此刻没地方放它们。</p>
            <p><strong>缺：</strong>包装 tool dispatch，把异常变成结构化 tool_result。<em>→ 第 6 章</em></p>
          </Card>

          <Card title="Break 4 · 模型永远停不下来" kind="bad">
            <p>mock 给 100 次相同 tool_call，<code>MAX_ITERATIONS=20</code> 在 20 步时抛 RuntimeError，丢掉所有 transcript。</p>
            <p>20 这个数字本身就是问题——对真任务太小，对真失控太大。<strong>需要的是基于成本的边界。</strong></p>
            <p><strong>缺：</strong>① token 预算（不是迭代次数）；② 保留 transcript 的可观测性；③ 检测重复调用的去重器。<em>→ 第 6 / 18 / 20 章</em></p>
          </Card>

          <Card title="Break 5 · 工具返回 200KB 的内容" kind="bad" style={{ gridColumn: "1 / -1" }}>
            <p><code>return "X" * 200_000</code>（想象一个 read_file 工具读了 60K-token 的 JSON）</p>
            <p>Loop 技术上还在跑——但 transcript 里塞进了 200KB，下一回合把整个 transcript 又发回 provider。第 5 回合早就超出窗口，provider 返回 error，会话神秘地崩——<strong>而真正的原因是没人在追踪工具返回了多少</strong>。</p>
            <p><strong>这是塑造全书后半的中心问题。</strong></p>

          </Card>
        </div>
