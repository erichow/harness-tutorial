        <p>最简单的主动理解工具：定期扫描项目目录，构建一棵树状索引。类似于第 24 章的 <code>list_directory</code>，但递归、缓存、带有文件大小统计。</p>

        <CodeBlock filename="src/harness/context/code-context.ts" lang="typescript">{`import { readdir, stat } from "node:fs/promises";
import { join, relative, resolve } from "node:path";

interface TreeNode {
  name: string;
  type: "file" | "dir";
  size?: number;
  children?: TreeNode[];
}

export class CodeContextManager {
  private projectRoot: string;
  private tree: TreeNode | null = null;
  private lastBuilt: number = 0;
  private readonly cacheTTLMs: number;

  constructor(root?: string, cacheTTLMs: number = 30_000) {
    this.projectRoot = resolve(root || process.cwd());
    this.cacheTTLMs = cacheTTLMs;
  }

  async getProjectTree(maxDepth: number = 3): Promise<TreeNode> {
    if (this.tree && Date.now() - this.lastBuilt < this.cacheTTLMs) {
      return this.tree;
    }

    const skipDirs = new Set([
      "node_modules", ".git", "dist", "build", ".next",
      "__pycache__", ".venv", "venv", ".turbo", ".cache",
    ]);

    const skipFiles = new Set([
      ".DS_Store", "package-lock.json", "yarn.lock", "pnpm-lock.yaml",
    ]);

    const buildTree = async (dir: string, depth: number): Promise<TreeNode> => {
      const name = relative(this.projectRoot, dir) || ".";
      const entries = await readdir(dir, { withFileTypes: true });

      const children: TreeNode[] = [];
      for (const entry of entries) {
        if (skipDirs.has(entry.name) || skipFiles.has(entry.name)) continue;

        const fullPath = join(dir, entry.name);
        if (entry.isDirectory()) {
          if (depth > 0) {
            children.push(await buildTree(fullPath, depth - 1));
          } else {
            // 超过深度时只标记目录名，不展开
            children.push({ name: entry.name + "/", type: "dir" });
          }
        } else {
          const info = await stat(fullPath);
          children.push({
            name: entry.name,
            type: "file",
            size: info.size,
          });
        }
      }

      children.sort((a, b) => {
        if (a.type !== b.type) return a.type === "dir" ? -1 : 1;
        return a.name.localeCompare(b.name);
      });

      return { name, type: "dir", children };
    };

    this.tree = await buildTree(this.projectRoot, maxDepth);
    this.lastBuilt = Date.now();
    return this.tree;
  }

  /** 生成适合模型理解的文本化结构 */
  async formatProjectTree(maxDepth: number = 3): Promise<string> {
    const tree = await this.getProjectTree(maxDepth);
    return this.formatNode(tree, "");
  }

  private formatNode(node: TreeNode, indent: string): string {
    if (node.type === "file") {
      const sizeStr = node.size ? \` (\${formatSize(node.size)})\` : "";
      return indent + node.name + sizeStr;
    }

    const lines: string[] = [indent + (node.name === "." ? "" : node.name + "/")];
    if (node.children) {
      for (const child of node.children) {
        lines.push(this.formatNode(child, indent + "  "));
      }
    }
    return lines.join("\n");
  }
}

function formatSize(bytes: number): string {
  if (bytes < 1024) return \`\${bytes}B\`;
  if (bytes < 1024 * 1024) return \`\${(bytes / 1024).toFixed(0)}KB\`;
  return \`\${(bytes / (1024 * 1024)).toFixed(1)}MB\`;
}`}</CodeBlock>

        <p>从 import/require 语句中提取文件间的依赖关系，构建一个有向图。这是 agent 理解"修改这个文件会影响哪些文件"的关键工具。</p>

        <CodeBlock filename="src/harness/context/code-context.ts" lang="typescript">{`interface DepGraph {
  [file: string]: {
    imports: string[];    // 这个文件导入的文件
    importedBy: string[];  // 导入这个文件的文件
  };
}

export class CodeContextManager {
  // ... 前面的代码

  private depGraph: DepGraph | null = null;

  /** 扫描所有源文件，构建依赖图 */
  async buildDependencyGraph(): Promise<DepGraph> {
    if (this.depGraph) return this.depGraph;

    const graph: DepGraph = {};
    const srcFiles = await this.findSourceFiles();

    for (const file of srcFiles) {
      const fullPath = join(this.projectRoot, file);
      try {
        const content = await readFile(fullPath, "utf-8");
        const imports = this.extractImports(content, file);
        graph[file] = { imports, importedBy: [] };
      } catch {
        graph[file] = { imports: [], importedBy: [] };
      }
    }

    // 构建反向引用
    for (const [file, deps] of Object.entries(graph)) {
      for (const imp of deps.imports) {
        if (graph[imp]) {
          graph[imp].importedBy.push(file);
        }
      }
    }

    this.depGraph = graph;
    return graph;
  }

  /** 简单 import 提取——支持 ES module 和 CommonJS */
  private extractImports(content: string, filePath: string): string[] {
    const imports: string[] = [];
    const lines = content.split("\n");

    for (const line of lines) {
      // import ... from "X"
      const esMatch = line.match(/import\s+.*\s+from\s+["']([^"']+)["']/);
      if (esMatch) {
        const resolved = this.resolveModulePath(esMatch[1], filePath);
        if (resolved) imports.push(resolved);
      }

      // require("X")
      const cjsMatch = line.match(/require\s*\(\s*["']([^"']+)["']\s*\)/);
      if (cjsMatch) {
        const resolved = this.resolveModulePath(cjsMatch[1], filePath);
        if (resolved) imports.push(resolved);
      }
    }

    return imports;
  }

  /** 简易的模块解析——仅处理相对路径 */
  private resolveModulePath(specifier: string, fromFile: string): string | null {
    if (!specifier.startsWith(".") && !specifier.startsWith("/")) {
      return null; // 外部 package——跳过
    }

    // 简化：假设 specifier 是要找的 .ts/.tsx 文件
    const dir = dirname(join(this.projectRoot, fromFile));
    const resolved = resolve(dir, specifier);
    const extensions = [".ts", ".tsx", ".js", ".jsx", "/index.ts", "/index.tsx"];

    for (const ext of extensions) {
      const attempt = resolved + ext;
      if (existsSync(attempt)) {
        return relative(this.projectRoot, attempt);
      }
    }

    return null;
  }

  /** 回答：修改这个文件会影响哪些其他文件？ */
  async getAffectedFiles(filePath: string): Promise<string[]> {
    const graph = await this.buildDependencyGraph();
    const file = relative(this.projectRoot, resolve(this.projectRoot, filePath));

    // BFS 查找所有受影响的文件（直接和传递依赖）
    const affected: Set<string> = new Set();
    const queue = [file];
    while (queue.length > 0) {
      const current = queue.shift()!;
      const deps = graph[current];
      if (!deps) continue;

      for (const importer of deps.importedBy) {
        if (!affected.has(importer)) {
          affected.add(importer);
          queue.push(importer);
        }
      }
    }

    return Array.from(affected);
  }
}`}</CodeBlock>

let contextManager: CodeContextManager;

export function setCodeContextManager(mgr: CodeContextManager): void {
  contextManager = mgr;
}

export const extractCodeContextTool: Tool = {
  name: "extract_code_context",
  description: "在代码库中搜索一个符号（函数、类、变量）的定义和引用。返回定义文件和关键引用位置。",
  inputSchema: {
    type: "object",
    properties: {
      symbol: { type: "string", description: "要搜索的符号名称" },
      maxResults: { type: "number", description: "最大结果数（默认 10）" },
    },
    required: ["symbol"],
  },
  async handler(args: { symbol: string; maxResults?: number }): Promise<string> {
    // 使用 grep 风格搜索来定位符号
    const maxResults = Math.min(args.maxResults ?? 10, 50);
    const root = process.cwd();
    const results: Array<{ file: string; line: number; text: string }> = [];
    const seenFiles = new Set<string>();

    // 递归搜索 src/ 目录

      const entries = await readdir(dir, { withFileTypes: true });
      for (const entry of entries) {
        if (entry.name === "node_modules" || entry.name === ".git" || entry.name.startsWith(".")) continue;
        const fullPath = join(dir, entry.name);

        if (entry.isDirectory()) {
          await searchDir(fullPath);
        } else if (/\.(ts|tsx|js|jsx|py|rs|go)$/.test(entry.name)) {
          const content = await readFile(fullPath, "utf-8");
          const lines = content.split("\n");
          for (let i = 0; i < lines.length; i++) {
            // 用正则查找符号定义（export/function/class/const + symbol）
            if (lines[i].includes(args.symbol)) {

                file: relative(root, fullPath),
                line: i + 1,
                text: lines[i].trim(),
              });
              if (results.length >= maxResults * 2) break;
            }
          }
        }
        if (results.length >= maxResults * 2) break;

        <CodeBlock filename="src/harness/context/code-context.ts" lang="typescript">{`interface BudgetAllocation {
  fullContent: string[];    // 需要全文的文件
  summaries: string[];       // 只需要摘要的文件
  skipped: string[];         // 跳过
}

export class CodeContextManager {
  /** 为给定任务分配上下文预算 */
  async allocateBudget(
    taskDescription: string,
    budgetTokens: number = 20_000,
  ): Promise<BudgetAllocation> {
    const tree = await this.getProjectTree();
    const graph = await this.buildDependencyGraph();
    const files = await this.collectAllFiles();

    // 角色优先级
    const rolePriority: Record<FileRole, number> = {
      entrypoint: 5,
      config: 4,
      "business-logic": 3,
      schema: 3,
      utility: 2,
      "ui-component": 2,
      test: 1,
      style: 1,
      documentation: 1,
      unknown: 1,
    };

    // 按角色 + 文件大小排序
    const scored = files.map(f => ({
      path: f,
      role: classifyFile(f),
      score: rolePriority[classifyFile(f)] ?? 1,
      // 被很多文件依赖的得分更高
      deps: (graph[f]?.importedBy?.length ?? 0),
    }));

    // 进入 top-K 的给全文，其他的给摘要
    scored.sort((a, b) => (b.score + b.deps) - (a.score + a.deps));

    const fullContent: string[] = [];
    const summaries: string[] = [];
    const skipped: string[] = [];
    let usedBudget = 0;

    for (const file of scored) {
      if (usedBudget >= budgetTokens) {
        skipped.push(file.path);
        continue;
      }

      const estimatedTokens = await this.estimateTokens(file.path);
      if (usedBudget + estimatedTokens <= budgetTokens) {
        fullContent.push(file.path);
        usedBudget += estimatedTokens;
      } else if (usedBudget + Math.ceil(estimatedTokens * 0.2) <= budgetTokens) {
        summaries.push(file.path);
        usedBudget += Math.ceil(estimatedTokens * 0.2);
      } else {
        skipped.push(file.path);
      }
    }

    return { fullContent, summaries, skipped };
  }

  private async estimateTokens(filePath: string): Promise<number> {
    try {
      const content = await readFile(join(this.projectRoot, filePath), "utf-8");
      // 粗略估算：1 token ≈ 4 字符
      return Math.ceil(content.length / 4);
    } catch {
      return 0;
    }
  }
}`}</CodeBlock>
