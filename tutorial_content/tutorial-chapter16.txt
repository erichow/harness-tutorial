/* ─── Interactive plan demo ─── */
function PlanInteractiveDemo() {
  const [steps, setSteps] = useS16([
    { desc: "Check file /etc/hostname exists",                status: "pending", evidence: "" },
    { desc: "Check file /etc/os-release exists",              status: "pending", evidence: "" },
    { desc: "Check file /etc/machine-id exists",              status: "pending", evidence: "" },
  ]);
  const [posts, setPosts] = useS16([
    { desc: "All three file paths + sizes reported",          satisfied: false, evidence: "" },
    { desc: "Winner (largest file) identified by name",       satisfied: false, evidence: "" },
  ]);
  const [tryFinal, setTryFinal] = useS16(false);

  const allStepsTerminal = steps.every(s => s.status === "done" || s.status === "blocked");
  const allPostsSatisfied = posts.every(p => p.satisfied);
  const readyToFinalize = allStepsTerminal && allPostsSatisfied;

  const toggleStep = (i) => {
    setSteps(prev => prev.map((s, idx) => {
      if (idx !== i) return s;
      const cycle = { pending: "in_progress", in_progress: "done", done: "blocked", blocked: "pending" };
      const next = cycle[s.status];
      const ev = next === "done" ? "(verified via bash test -f)" : "";
      return { ...s, status: next, evidence: ev };
    }));
  };
  const togglePost = (i) => {
    setPosts(prev => prev.map((p, idx) => idx === i
      ? { ...p, satisfied: !p.satisfied, evidence: !p.satisfied ? "(matches step evidence)" : "" }
      : p));
  };

  const statusBadge = (s) => {
    const cfg = {
      pending:     ["[ ]", "var(--ink-soft)"],
      in_progress: ["[.]", "var(--accent-3)"],
      done:        ["[x]", "var(--green)"],
      blocked:     ["[!]", "var(--red)"],
    }[s];
    return <span style={{ fontFamily: "var(--font-mono)", color: cfg[1], fontWeight: 700 }}>{cfg[0]}</span>;
  };

  return (
    <div className="loop-demo">
      <div style={{ fontFamily: "var(--font-h)", fontSize: 24, marginBottom: 6 }}>
        🎛 自己试试：点状态切换它的进度
      </div>
      <div style={{ fontFamily: "var(--font-hand)", fontSize: 15, color: "var(--ink-soft)", marginBottom: 14 }}>
        Plan 是一个 <em>harness 强制</em>的结构化对象。点 step / postcondition 状态切换，看 final gate 是否开门。
      </div>

      <div style={{ marginBottom: 14 }}>
        <div style={{ fontFamily: "var(--font-h)", fontSize: 20, color: "var(--ink)", marginBottom: 8 }}>
          # Plan: 验证三个文件并报告最大者
        </div>

        <div style={{ fontFamily: "var(--font-h)", fontSize: 17, color: "var(--accent-4)", marginTop: 8 }}>
          Steps
        </div>
        {steps.map((s, i) => (
          <div key={i} style={{ display: "flex", alignItems: "center", gap: 10,
                                padding: "6px 0", fontFamily: "var(--font-hand)", fontSize: 15 }}>
            <button onClick={() => toggleStep(i)}
                    style={{ background: "var(--paper)", border: "1.5px solid var(--rule)",
                             padding: "3px 8px", borderRadius: 4, cursor: "pointer",
                             fontFamily: "var(--font-mono)", fontSize: 14, minWidth: 40 }}>
              {statusBadge(s.status)}
            </button>
            <span style={{ flex: 1 }}>{i + 1}. {s.desc}</span>
            {s.evidence && (
              <span style={{ fontSize: 13, color: "var(--green)", fontStyle: "italic" }}>
                evidence: {s.evidence}
              </span>
            )}
          </div>
        ))}

        <div style={{ fontFamily: "var(--font-h)", fontSize: 17, color: "var(--accent-4)", marginTop: 14 }}>
          Postconditions
        </div>
        {posts.map((p, i) => (
          <div key={i} style={{ display: "flex", alignItems: "center", gap: 10,
                                padding: "6px 0", fontFamily: "var(--font-hand)", fontSize: 15 }}>
            <button onClick={() => togglePost(i)}
                    style={{ background: "var(--paper)", border: "1.5px solid var(--rule)",
                             padding: "3px 8px", borderRadius: 4, cursor: "pointer",
                             fontFamily: "var(--font-mono)", fontSize: 14, minWidth: 40 }}>
              {p.satisfied
                ? <span style={{ color: "var(--green)", fontWeight: 700 }}>[x]</span>
                : <span style={{ color: "var(--ink-soft)" }}>[ ]</span>}
            </button>
            <span style={{ flex: 1 }}>{i + 1}. {p.desc}</span>
            {p.evidence && (
              <span style={{ fontSize: 13, color: "var(--green)", fontStyle: "italic" }}>
                evidence: {p.evidence}
              </span>
            )}
          </div>
        ))}
      </div>

      {/* Try-finalize button */}
      <div style={{ display: "flex", gap: 14, alignItems: "center", marginTop: 12,
                    padding: 14, borderTop: "1.5px dashed var(--rule)" }}>
        <button onClick={() => setTryFinal(true)}
                style={{ background: "var(--accent)", color: "var(--paper)",
                         border: "2px solid var(--ink)", padding: "8px 18px",
                         borderRadius: 999, fontFamily: "var(--font-h)", fontSize: 18,
                         cursor: "pointer", fontWeight: 700 }}>
          模型: "我做完了！"
        </button>
        {tryFinal && (
          readyToFinalize
            ? <span style={{ color: "var(--green)", fontFamily: "var(--font-h)", fontSize: 20, fontWeight: 700 }}>
                ✓ Gate 通过 — final answer 被接受
              </span>
            : <div style={{ flex: 1, color: "var(--red)", fontFamily: "var(--font-hand)", fontSize: 14 }}>
                <strong>✗ Harness 拒绝 finalization：</strong>
                <br />Plan is not complete. Before declaring the task done, either mark remaining steps
                as done with evidence, verify outstanding postconditions, or mark them blocked with a reason.
              </div>
        )}
      </div>
    </div>
  );
}

    pending = "pending"
    in_progress = "in_progress"
    done = "done"
    blocked = "blocked"


@dataclass
class Step:
    id: str
    description: str
    status: StepStatus = StepStatus.pending
    evidence: str | None = None   # 什么证明了 step 完成
    notes: str = ""

    def is_terminal(self) -> bool:
        return self.status in (StepStatus.done, StepStatus.blocked)


@dataclass
class Postcondition:
    description: str
    satisfied: bool = False
    evidence: str | None = None


@dataclass
class Plan:
    objective: str
    steps: list[Step] = field(default_factory=list)
    postconditions: list[Postcondition] = field(default_factory=list)
    created_at: datetime = field(default_factory=lambda: datetime.now(timezone.utc))
    id: str = field(default_factory=lambda: str(uuid4()))

    def all_steps_terminal(self) -> bool:
        return all(s.is_terminal() for s in self.steps)

    def all_postconditions_satisfied(self) -> bool:
        return all(pc.satisfied for pc in self.postconditions)

    def is_ready_to_finalize(self) -> bool:
        return self.all_steps_terminal() and self.all_postconditions_satisfied()

    def to_render(self) -> str:
        """Render the plan as a string the model can read."""
        lines = [f"# Plan: {self.objective}\\n"]
        lines.append("## Steps")
        for i, s in enumerate(self.steps, start=1):
            mark = {"pending": "[ ]", "in_progress": "[.]",
                    "done": "[x]", "blocked": "[!]"}[s.status.value]
            lines.append(f"{i}. {mark} {s.description}")
            if s.evidence: lines.append(f"   evidence: {s.evidence}")
            if s.notes:    lines.append(f"   notes: {s.notes}")
        lines.append("\\n## Postconditions")
        for i, pc in enumerate(self.postconditions, start=1):
            mark = "[x]" if pc.satisfied else "[ ]"
            lines.append(f"{i}. {mark} {pc.description}")
            if pc.evidence: lines.append(f"   evidence: {pc.evidence}")
        return "\\n".join(lines)`}</CodeBlock>

        <Callout kind="key" title="两个值得说的区分">

class PlanHolder:
    """Wraps a mutable Plan so tools can mutate it through a shared reference."""
    plan: Plan | None = None

    def require(self) -> Plan:
        if self.plan is None:
            raise RuntimeError("no active plan")
        return self.plan


def plan_tools(holder: PlanHolder) -> list[Tool]:

    @tool(side_effects={"write"})
    def plan_create(objective: str, steps: list[str], postconditions: list[str]) -> str:
        """Create or replace the plan for this session.

        objective: 一句话描述你要完成什么。
        steps: 有序的 step 描述列表。每条是具体可行动的，不是模糊意图。
        postconditions: 任务声称完成时必须为真的条件。
                        例："文件 X 存在且含 Y"；"模块 Z 的测试通过"。

        Call this once at the start of any non-trivial task, before beginning work.
        If the plan is wrong mid-task, call this again to replace it — the harness
        records the rewrite.
        """
        holder.plan = Plan(
            objective=objective,
            steps=[Step(id=f"s{i}", description=d) for i, d in enumerate(steps)],
            postconditions=[Postcondition(description=d) for d in postconditions],
        )
        return f"plan created with {len(steps)} steps and {len(postconditions)} postconditions"

    @tool(side_effects={"read"})
    def plan_show() -> str:
        """Display the current plan with status."""
        return holder.require().to_render()

    @tool(side_effects={"write"})
    def step_update(step_number: int, status: str,
                    evidence: str = "", notes: str = "") -> str:
        """Update a step's status.

        evidence: required for 'done'. 一句话证明完成。
        """
        plan = holder.require()
        if step_number < 1 or step_number > len(plan.steps):
            return f"step_number {step_number} out of range (1..{len(plan.steps)})"
        try:
            new_status = StepStatus(status)
        except ValueError:
            return f"invalid status {status!r}; use pending/in_progress/done/blocked"

        if new_status == StepStatus.done and not evidence:
            return ("error: marking a step 'done' requires evidence. Describe "
                    "what proved the step complete (e.g., 'wrote file and "
                    "read it back; content matches').")

        s = plan.steps[step_number - 1]
        plan.steps[step_number - 1] = Step(
            id=s.id, description=s.description,
            status=new_status, evidence=evidence or None, notes=notes,
        )
        return f"step {step_number} → {status}"

    @tool(side_effects={"write"})
    def postcondition_verify(postcondition_number: int, evidence: str) -> str:
        """Mark a postcondition as verified.

        evidence: required. Concrete proof the postcondition holds.
        This is what the harness checks before letting you declare the task complete.
        """
        plan = holder.require()
        if postcondition_number < 1 or postcondition_number > len(plan.postconditions):
            return f"postcondition_number out of range"
        if not evidence:
            return "error: evidence is required to verify a postcondition"
        pc = plan.postconditions[postcondition_number - 1]
        plan.postconditions[postcondition_number - 1] = Postcondition(
            description=pc.description, satisfied=True, evidence=evidence,
        )
        return f"postcondition {postcondition_number} verified"

    return [plan_create, plan_show, step_update, postcondition_verify]`}</CodeBlock>

        <Callout kind="key" title="两个值得观察的行为">
          <p><strong>开头</strong>，agent 调 <code>plan_create</code>，3 个 step（每文件一个）+ 3 个 postcondition。<em>没有 plan_holder 机制，粗心的 agent 可能检查两个文件就说完成</em>。有了它，如果 agent 跳过 step 3 想 finalize，<strong>harness 拒绝 final answer，agent 回去检查第 3 个文件</strong>。</p>

          </Card>

            <br /><br />阻止会让 agent 变脆——plan 当初错了时合理需要修改。<em>正确的 audit 是观测，不是 enforcement</em>。
