let progressLine = "";

case "progress":
  // 清除上一行进度条
  if (progressLine) {
    process.stdout.write("\\r" + " ".repeat(progressLine.length) + "\\r");
  }
  const bar = renderProgressBar(event.progress ?? 0);
  progressLine = \`\${bar} \${event.message}\`;

  const empty = width - filled;
  return \`[\${"█".repeat(filled)}\${"░".repeat(empty)}] \${percent}%\`;

export const confirmDialogTool: Tool = {
  name: "confirm_dialog",
  description: "向用户显示一个确认对话框，等待用户响应。适用于重要的操作前获取用户同意。",
  inputSchema: {
    type: "object",
    properties: {
      message: {
        type: "string",
        description: "向用户展示的确认问题",
      },
      options: {
        type: "array",
        items: { type: "string" },
        description: "可选选项（默认 ['y', 'n']）",
      },
      defaultResponse: {
        type: "string",
        description: "默认响应（按 Enter 时的选择）",
      },
    },
    required: ["message"],
  },
  async handler(args: { message: string; options?: string[]; defaultResponse?: string }): Promise<string> {
    const opts = args.options ?? ["y", "n"];
    const defaultResp = args.defaultResponse ?? opts[0];
    const prompt = \`\n\${args.message} (\${opts.join("/")}) [\${defaultResp}]: \`;

    const rl = createInterface({
      input: process.stdin,
      output: process.stdout,
    });

    try {
      const answer = await rl.question(prompt);
      const trimmed = answer.trim().toLowerCase();
      if (trimmed === "") return defaultResp;
      return trimmed;
    } finally {
      rl.close();
    }
  },
};

export const inputDialogTool: Tool = {
  name: "input_dialog",

  inputSchema: {
    type: "object",
    properties: {
      message: {
        type: "string",
        description: "向用户展示的输入提示",
      },
      defaultValue: {
        type: "string",
        description: "可选的默认值",
      },
      multiline: {
        type: "boolean",
        description: "是否允许多行输入（默认 false）",
      },
    },
    required: ["message"],
  },
  async handler(
    args: { message: string; defaultValue?: string; multiline?: boolean },

        <p>文件编辑是 coding agent 最核心也最危险的操作。一个好的 diff 视图让用户在变更执行前看到改动——能显著提升信任度。</p>

        <CodeBlock filename="src/harness/tools/ui.ts" lang="typescript">{`export const showDiffViewTool: Tool = {
  name: "show_diff_view",
  description: "显示文件变更的差异对比。在文件编辑前调用，让用户预览即将发生的变化。",
  inputSchema: {
    type: "object",
    properties: {
      filePath: {
        type: "string",
        description: "要展示 diff 的文件路径",
      },
      originalSnippet: {
        type: "string",
        description: "原始内容片段",
      },
      modifiedSnippet: {
        type: "string",
        description: "修改后的内容片段",
      },
      description: {
        type: "string",
        description: "变更的简要说明",
      },
    },
    required: ["filePath", "originalSnippet", "modifiedSnippet"],
  },
  handler(args: {
    filePath: string;
    originalSnippet: string;
    modifiedSnippet: string;
    description?: string;
  }): string {
    const origLines = args.originalSnippet.split("\n");
    const modLines = args.modifiedSnippet.split("\n");

    // 生成一个简化的 unified diff 格式
    const diffLines: string[] = [
      \`\n--- 变更: \${args.filePath}\`,
      args.description ? \`    说明: \${args.description}\` : "",
      "┌─ 变更对比 ─────────────────────────────",
    ];

    const maxLines = Math.max(origLines.length, modLines.length);
    for (let i = 0; i < maxLines; i++) {
      const orig = i < origLines.length ? origLines[i] : "";
      const mod = i < modLines.length ? modLines[i] : "";

      if (orig === mod) {
        diffLines.push(\`│  \${orig}\`);
      } else {
        if (orig) diffLines.push(\`│ -\${orig}\`);
        if (mod) diffLines.push(\`│ +\${mod}\`);
      }
    }

    diffLines.push("└───────────────────────────────────");

    // 也通过 event emitter 发送，给前端渲染器处理
    if (eventEmitter) {
      eventEmitter.emit({
        type: "ui_diff",
        filePath: args.filePath,
        original: args.originalSnippet,
        modified: args.modifiedSnippet,
        timestamp: Date.now(),
      });
    }

    return diffLines.join("\n");
  },
};`}</CodeBlock>

      <Section id="ch29-5" num="29.5" title="会话内记忆——\u201C总是允许\u201D的持久化">
        <p>用户确认操作的"总是在本次会话中允许"决策需要在客户端持久化。实现很简单——一个 Map。</p>

        <CodeBlock filename="src/cli/permissions.ts" lang="typescript">{`// 会话内"总是允许"缓存
class SessionPermissionCache {
  private cache = new Map<string, boolean>();

  /** 检查是否已记忆 */
  isRemembered(action: string, details?: string): boolean {
    const key = this.buildKey(action, details);
    return this.cache.has(key);
  }

  /** 记忆一个决策 */
  remember(action: string, details: string | undefined, allowed: boolean): void {
    const key = this.buildKey(action, details);
    this.cache.set(key, allowed);
  }

  /** 获取缓存的决策 */
  get(action: string, details?: string): boolean | undefined {
    const key = this.buildKey(action, details);
    return this.cache.get(key);
  }

  /** 重置所有记忆 */
  reset(): void {
    this.cache.clear();
  }

  private buildKey(action: string, details?: string): string {
    // 对于工具调用，key 是 action + 截断的参数
    // 只缓存同一类操作，不缓存具体参数
    return \`\${action}::\${details?.slice(0, 100) ?? ""}\`;
  }
}

const sessionCache = new SessionPermissionCache();

export async function cliPermissionHandler(
  req: PermissionRequest,
): Promise<PermissionDecision> {
  // 检查是否已记忆
  const remembered = sessionCache.get(req.action, req.details);
  if (remembered !== undefined) {
    return { allowed: remembered, reason: "session-cached" };
  }

  const decision = await interactiveConfirm(req);

  if (decision.remember) {
    sessionCache.remember(req.action, req.details, decision.allowed);
  }

  return decision;
}

// 重置缓存——绑定到"不安全"命令或用户显式请求
export function resetPermissionCache(): void {
  sessionCache.reset();
  console.log("🔄 权限记忆已重置。后续所有操作将重新确认。");
}`}</CodeBlock>

        <Callout kind="key" title="\u201C总是允许\u201D的范围很重要">
          <p>"总是允许"应该绑定到<strong>操作类</strong>而不是具体命令。例如，用户允许 <code>run_command: ls -la</code>，不应该自动允许 <code>run_command: rm -rf /</code>。<br />
          但 <code>run_command: ls src/</code> 和 <code>run_command: ls tests/</code> 应该共享同一个"总是允许"决策吗？这取决于你的安全策略。本章的实现以 action + 前 100 字符摘要作为 key，是一个合理的平衡点。</p>
        </Callout>
      </Section>

        <p>Claude Code 有一个"思考模式"——模型在给出答案前先展示推理过程。这可以通过 StreamEvent 实现：</p>

        <CodeBlock filename="src/cli/events.ts" lang="typescript">{`// 在 cliEventHandler 中添加思考模式渲染
let inThinkingBlock = false;

case "thinking":
  if (!inThinkingBlock) {
    process.stdout.write("\n🤔 ");
    inThinkingBlock = true;
  }
  process.stdout.write(event.delta);
  break;

case "thinking_end":
  if (inThinkingBlock) {
    process.stdout.write("\n\n");
    inThinkingBlock = false;
  }
  break;

case "text":
  // 如果刚退出思考模式，加一个分隔
  if (inThinkingBlock) {
    process.stdout.write("\n\n");
    inThinkingBlock = false;
  }
  process.stdout.write(event.delta);
  break;`}</CodeBlock>
