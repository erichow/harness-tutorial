import { type Tool } from "./types.js";
import { resolve } from "node:path";

// 每个工具都从同一个 git 实例获取
let gitInstance: SimpleGit | null = null;

function getGit(cwd?: string): SimpleGit {
  if (!gitInstance) {
    gitInstance = simpleGit({ baseDir: cwd ?? process.cwd() });
  }
  return gitInstance;
}

export const gitStatusTool: Tool = {
  name: "git_status",
  description: "显示当前 Git 仓库的状态：分支、未提交的变更、未跟踪的文件。",
  inputSchema: {
    type: "object",
    properties: {
      path: { type: "string", description: "Git 仓库路径（默认当前目录）" },
    },
  },
  async handler(args: { path?: string }): Promise<string> {
    const git = getGit(args.path);
    const status: StatusResult = await git.status();

    const lines: string[] = [
      \`分支: \${status.current}\`,
      \`未暂存的更改: \${status.modified.length} 个文件\`,
      \`已暂存的更改: \${status.staged.length} 个文件\`,
      \`未跟踪的文件: \${status.not_added.length} 个文件\`,
    ];

    if (status.modified.length > 0) {
      lines.push("\\n已修改（未暂存）:");
      status.modified.forEach(f => lines.push(\`  ✏️ \${f}\`));
    }
    if (status.staged.length > 0) {
      lines.push("\\n已暂存（待提交）:");
      status.staged.forEach(f => lines.push(\`  ✅ \${f}\`));
    }
    if (status.not_added.length > 0) {
      lines.push("\\n未跟踪:");
      status.not_added.forEach(f => lines.push(\`  ❓ \${f}\`));
    }

    return lines.join("\n");
  },
};

export const gitDiffTool: Tool = {
  name: "git_diff",
  description: "显示文件差异。默认显示所有未暂存的修改，或指定文件的差异。",
  inputSchema: {
    type: "object",
    properties: {
      path: { type: "string", description: "可选，只显示特定文件的差异" },
      staged: { type: "boolean", description: "是否显示已暂存（staged）的差异" },
    },
  },
  async handler(args: { path?: string; staged?: boolean }): Promise<string> {
    const git = getGit();

    const options: string[] = [];
    if (args.staged) options.push("--staged");
    if (args.path) options.push("--", args.path);

    const diff = await git.diff(options);

    if (!diff.trim()) {
      return "没有差异。";
    }

    // Diff 可能很大——截断到合理长度
    const MAX_DIFF = 50_000;
    if (diff.length > MAX_DIFF) {
      return diff.slice(0, MAX_DIFF) + \`\\n…(差异截断，共 \${diff.length} 字符)\`;
    }

    return diff;
  },
};

export const gitLogTool: Tool = {
  name: "git_log",
  description: "显示 Git 提交历史。可限制条数和文件范围。",
  inputSchema: {
    type: "object",
    properties: {
      maxCount: { type: "number", description: "最大提交数（默认 10）" },
      path: { type: "string", description: "可选，只显示特定文件的提交历史" },
    },
  },
  async handler(args: { maxCount?: number; path?: string }): Promise<string> {
    const git = getGit();
    const log = await git.log({
      maxCount: Math.min(args.maxCount ?? 10, 50),
      ...(args.path ? { file: args.path } : {}),
    });

    if (log.total === 0) {
      return "没有提交历史。";
    }

    return log.all.map(commit =>
      \`\${commit.hash.slice(0, 8)} \${commit.date.slice(0, 10)} \${commit.message}\`
    ).join("\n");
  },
};

/* ──────────── 写操作：需确认 ──────────── */

export const gitCommitTool: Tool = {
  name: "git_commit",
  description: "暂存并提交所有更改。提供清晰的提交信息。需要用户确认。",
  inputSchema: {
    type: "object",
    properties: {
      message: {
        type: "string",
        description: "提交信息。注意：会暂存所有已跟踪文件的修改（相当于 git add -A 然后 git commit）",
      },
    },
    required: ["message"],
  },
  async handler(args: { message: string }): Promise<string> {
    const git = getGit();

    // 先把当前状态发给用户审核用
    const status = await git.status();
    const stagedSummary = [
      \`修改: \${status.modified.length} 个文件\`,
      \`删除: \${status.deleted.length} 个文件\`,
      \`新增: \${status.not_added.length} 个文件\`,
    ].join("，");

    // 用户已经通过 permission 层确认过，这里直接执行
    await git.add(".");
    await git.commit(args.message);

    return \`提交成功: \${args.message}
  影响: \${stagedSummary}\`;
  },
};

export const gitStashTool: Tool = {
  name: "git_stash",
  description: "暂存当前更改（git stash），恢复工作区到干净状态。",
  inputSchema: {
    type: "object",
    properties: {
      message: { type: "string", description: "可选的暂存说明" },
      action: {
        type: "string",
        enum: ["push", "pop", "list"],
        description: "push=暂存更改, pop=恢复最近暂存, list=查看所有暂存",
      },
    },
    required: ["action"],
  },
  async handler(args: { message?: string; action: string }): Promise<string> {
    const git = getGit();

    switch (args.action) {
      case "push":
        await git.stash([...(args.message ? ["-m", args.message] : [])]);
        return "更改已暂存 (stash)。可用 git_stash action=pop 恢复。";
      case "pop":
        await git.stash(["pop"]);
        return "最近一次暂存已恢复。";
      case "list":
        const list = await git.stashList();
        if (list.all.length === 0) return "没有暂存的更改。";
        return list.all.map(s =>
          \`\${s.hash.slice(0, 8)}: \${s.message}\`
        ).join("\n");
      default:
        return \`未知操作: \${args.action}\`;
    }
  },
};

export const gitBranchTool: Tool = {
  name: "git_branch",
  description: "创建、切换、删除分支。列出所有分支时不需要确认。",
  inputSchema: {
    type: "object",
    properties: {
      action: {
        type: "string",
        enum: ["list", "create", "switch", "delete"],
        description: "list=列出、create=创建、switch=切换、delete=删除",
      },
      name: {
        type: "string",
        description: "分支名（create/switch/delete 时需要）",
      },
    },
    required: ["action"],
  },
  async handler(args: { action: string; name?: string }): Promise<string> {
    const git = getGit();

    switch (args.action) {
      case "list":
        const branches = await git.branch();
        return branches.all.map(b => {
          const current = b === branches.current ? "* " : "  ";
          return current + b;
        }).join("\n");

      case "create":
        if (!args.name) return "错误：create 操作需要 name 参数";
        await git.checkoutLocalBranch(args.name);
        return \`已创建并切换到分支: \${args.name}\`;

      case "switch":
        if (!args.name) return "错误：switch 操作需要 name 参数";
        await git.checkout(args.name);
        return \`已切换到分支: \${args.name}\`;

      case "delete":
        if (!args.name) return "错误：delete 操作需要 name 参数";
        await git.deleteLocalBranch(args.name);
        return \`已删除分支: \${args.name}\`;

      default:
        return \`未知操作: \${args.action}\`;
    }
  },
};`}</CodeBlock>

export async function cliPermissionHandler(req: PermissionRequest): Promise<PermissionDecision> {
  // Git 读操作自动放行
