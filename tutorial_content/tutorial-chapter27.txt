        <p>LSP 是 JSON-RPC 之上的协议，客户端与语言服务器之间通过 stdin/stdout（或 TCP）通信。</p>

        <CodeBlock lang="text">{`┌──────────┐         JSON-RPC          ┌──────────────┐
│  Client   │ ──── initialize ──────→  │  Language    │
│ (editor   │ ←── capabilities ──────  │  Server      │
│  / agent) │                           │              │
│           │ ──── didOpen(text) ───→  │  (ts-server, │
│           │ ──── completion ──────→  │  pyright,    │
│           │ ←── CompletionList ────  │  rust-analyzer)│

      <Section id="ch27-2" num="27.2" title="初始化与生命周期管理">
        <p>实现一个 LSP 客户端管理器：启动服务器、初始化、维护连接、关闭。</p>

        <CodeBlock filename="src/harness/lsp/client.ts" lang="typescript">{`import { spawn, type ChildProcess } from "node:child_process";
import { createInterface, type Interface } from "node:readline";
import { EventEmitter } from "node:events";

interface LSPServerConfig {
  command: string;
  args: string[];
  language: string;
}

interface LSPResponse {
  id: number;
  result?: unknown;
  error?: { code: number; message: string };
}

export class LSPClient extends EventEmitter {
  private process: ChildProcess | null = null;
  private rl: Interface | null = null;
  private requestId = 1;
  private pending = new Map<number, {
    resolve: (v: unknown) => void;
    reject: (e: Error) => void;
  }>();
  private buffer = "";
  private initialized = false;
  private capabilities: Record<string, unknown> = {};
  readonly language: string;

  constructor(readonly config: LSPServerConfig) {
    super();
    this.language = config.language;
  }

  async start(): Promise<void> {
    this.process = spawn(this.config.command, this.config.args, {
      stdio: ["pipe", "pipe", "pipe"],
    });

    this.rl = createInterface({ input: this.process.stdout! });
    this.rl.on("line", (line) => this.handleMessage(line));
    this.process.stderr!.on("data", (chunk: Buffer) => {
      // 很多 LSP 服务器把日志输出到 stderr
      this.emit("log", chunk.toString());
    });
    this.process.on("exit", (code) => {
      this.emit("exit", code);
      this.initialized = false;
    });

    // 发送 initialize 请求
    const result = await this.request("initialize", {
      processId: process.pid,
      capabilities: {
        textDocument: {
          completion: { completionItem: { snippetSupport: true } },
          definition: {},
          references: {},
          hover: {},
          diagnostic: {},
        },
      },
      workspaceFolder: null,
    });

    this.capabilities = result as Record<string, unknown>;
    this.initialized = true;

    // 通知服务器 initialized
    this.notify("initialized", {});
  }

  async openDocument(uri: string, text: string): Promise<void> {
    this.notify("textDocument/didOpen", {
      textDocument: {
        uri,
        languageId: this.language,
        version: 1,
        text,
      },
    });
  }

  async closeDocument(uri: string): Promise<void> {
    this.notify("textDocument/didClose", {
      textDocument: { uri },
    });
  }

  async getCompletion(uri: string, line: number, column: number): Promise<unknown> {
    return this.request("textDocument/completion", {
      textDocument: { uri },
      position: { line, character: column },
    });
  }

  async getDefinition(uri: string, line: number, column: number): Promise<unknown> {
    return this.request("textDocument/definition", {
      textDocument: { uri },
      position: { line, character: column },
    });
  }

  async getReferences(uri: string, line: number, column: number): Promise<unknown> {
    return this.request("textDocument/references", {
      textDocument: { uri },
      position: { line, character: column },
      context: { includeDeclaration: true },
    });
  }

  async getHover(uri: string, line: number, column: number): Promise<unknown> {
    return this.request("textDocument/hover", {
      textDocument: { uri },
      position: { line, character: column },
    });
  }

  async getDiagnostics(uri: string): Promise<unknown> {
    return this.request("textDocument/diagnostic", {
      textDocument: { uri },
    });
  }

  async shutdown(): Promise<void> {
    if (!this.initialized) return;
    await this.request("shutdown", null);
    this.notify("exit", {});
    this.process?.kill();
  }

  private async request(method: string, params: unknown): Promise<unknown> {
    const id = this.requestId++;
    const message = JSON.stringify({
      jsonrpc: "2.0",
      id,
      method,
      params,
    });

    return new Promise((resolve, reject) => {
      this.pending.set(id, { resolve, reject });
      this.send(message);
    });
  }

  private notify(method: string, params: unknown): void {
    const message = JSON.stringify({
      jsonrpc: "2.0",
      method,
      params,
    });
    this.send(message);
  }

  private send(message: string): void {
    // LSP 使用 Header（Content-Length） + 换行 + Body 的格式
    const header = \`Content-Length: \${Buffer.byteLength(message, "utf-8")}\\r\\n\`;
    this.process?.stdin?.write(header + "\\r\\n" + message);
  }

  private handleMessage(line: string): void {
    // LSP 消息是 Content-Length header + 空行 + JSON body
    // 内容长度已在 header 中，我们需要积累 buffer 直到拿到完整消息
    this.buffer += line + "\\n";

    // 检查是否有完整的 LSP 消息
    const match = this.buffer.match(/Content-Length: (\\d+)\\r\\n\\r\\n/);
    if (!match) return;

    const contentLength = parseInt(match[1], 10);
    const headerEnd = this.buffer.indexOf("\\r\\n\\r\\n") + 4;
    const body = this.buffer.slice(headerEnd);

    if (body.length < contentLength) return; // 还没收到完整 body

    this.buffer = this.buffer.slice(headerEnd + contentLength);

    try {
      const msg = JSON.parse(body) as LSPResponse & { method?: string };

      if (msg.id !== undefined && this.pending.has(msg.id)) {
        const { resolve, reject } = this.pending.get(msg.id)!;
        this.pending.delete(msg.id);

        if (msg.error) {
          reject(new Error(\`LSP error: \${msg.error.message}\`));
        } else {
          resolve(msg.result);
        }
      } else if (msg.method) {
        // 服务端主动通知（如 diagnostics）
        this.emit("notification", msg.method, msg.params);
      }
    } catch (err) {
      this.emit("parseError", err, body);
    }
  }
}`}</CodeBlock>

        <Callout kind="warn" title="LSP 客户端实现的隐性成本">

        <p>项目通常包含多种语言。需要一个管理器来按需启动和复用 LSP 服务器。</p>

        <CodeBlock filename="src/harness/lsp/manager.ts" lang="typescript">{`import { LSPClient } from "./client.js";

interface LSPConfig {
  [language: string]: {
    command: string;
    args: string[];
  };
}

const DEFAULT_LSP_CONFIG: LSPConfig = {
  typescript: {
    command: "typescript-language-server",
    args: ["--stdio"],
  },
  javascript: {
    command: "typescript-language-server",
    args: ["--stdio"],
  },
  python: {
    command: "pyright-langserver",
    args: ["--stdio"],
  },
};

export class LSPManager {
  private clients = new Map<string, LSPClient>();
  private config: LSPConfig;

  constructor(config?: Partial<LSPConfig>) {
    this.config = { ...DEFAULT_LSP_CONFIG, ...config };
  }

  async getClient(language: string): Promise<LSPClient> {
    const existing = this.clients.get(language);
    if (existing) return existing;

    const cfg = this.config[language];
    if (!cfg) {
      throw new Error(\`Unsupported language: \${language}\`);
    }

    const client = new LSPClient({
      command: cfg.command,
      args: cfg.args,
      language,
    });

    await client.start();
    this.clients.set(language, client);
    return client;
  }

  async closeAll(): Promise<void> {
    for (const client of this.clients.values()) {
      await client.shutdown();
    }
    this.clients.clear();
  }
}`}</CodeBlock>

        <Callout kind="key">
          <p>选择统一入口的核心原因：<strong>工具悬崖（第 12 章）。</strong> 你已经有文件工具（6 个）、终端工具（5 个）、Git 工具（6 个）……再加 6 个 LSP 工具就是 23 个。在 20+ 工具的场景下，模型选错工具的概率大幅上升。一个统一的 <code>lsp_query</code> 工具加上 <code>operation</code> 参数，让模型只需要知道"需要 IDE 能力时用这个工具"，然后在参数里指定具体操作。</p>

        <CodeBlock filename="src/harness/tools/lsp.ts" lang="typescript">{`import { LSPManager } from "../lsp/manager.js";
import { type Tool } from "./types.js";
import { readFile } from "node:fs/promises";
import { resolve } from "node:path";

// 全局 LSP 管理器
const lspManager = new LSPManager();

// 语言 -> 文件后缀映射
const LANGUAGE_MAP: Record<string, string> = {
  ".ts": "typescript",
  ".tsx": "typescript",
  ".js": "javascript",
  ".jsx": "javascript",
  ".py": "python",
};

function detectLanguage(path: string): string | null {
  const ext = Object.entries(LANGUAGE_MAP)
    .find(([suffix]) => path.endsWith(suffix));
  return ext?.[1] ?? null;
}

function pathToUri(absolutePath: string): string {
  return "file://" + absolutePath;
}

export const lspQueryTool: Tool = {
  name: "lsp_query",
  description: "向语言服务器发送查询。支持 operation: completion（补全）、definition（跳转定义）、references（查找引用）、hover（悬停文档）、diagnostics（诊断错误）。需要提供文件路径和行/列号。",
  inputSchema: {
    type: "object",
    properties: {
      operation: {
        type: "string",
        enum: ["completion", "definition", "references", "hover", "diagnostics"],
        description: "要执行的 LSP 操作",
      },
      path: {
        type: "string",
        description: "目标文件路径（相对于项目根目录）",
      },
      line: {
        type: "number",
        description: "行号（从 0 开始，definition/references/hover/completion 需要）",
      },
      column: {
        type: "number",
        description: "列号（从 0 开始，definition/references/hover/completion 需要）",
      },
    },
    required: ["operation", "path"],
  },
  async handler(args: {
    operation: string;
    path: string;
    line?: number;
    column?: number;
  }): Promise<string> {
    const language = detectLanguage(args.path);
    if (!language) {
      return \`不支持的文件类型: \${args.path}\`;
    }

    const client = await lspManager.getClient(language);
    const absPath = resolve(process.cwd(), args.path);
    const uri = pathToUri(absPath);

    // 打开文档（如果不打开，LSP 服务器无法分析）
    const content = await readFile(absPath, "utf-8");
    await client.openDocument(uri, content);

    try {
      switch (args.operation) {
        case "completion": {
          if (args.line === undefined || args.column === undefined) {
            return "completion 需要 line 和 column 参数";
          }
          const result = await client.getCompletion(uri, args.line, args.column);
          return formatCompletion(result);
        }

        case "definition": {
          if (args.line === undefined || args.column === undefined) {
            return "definition 需要 line 和 column 参数";
          }
          const result = await client.getDefinition(uri, args.line, args.column);
          return formatLocations(result);
        }

        case "references": {
          if (args.line === undefined || args.column === undefined) {
            return "references 需要 line 和 column 参数";
          }
          const result = await client.getReferences(uri, args.line, args.column);
          return formatLocations(result);
        }

        case "hover": {
          if (args.line === undefined || args.column === undefined) {
            return "hover 需要 line 和 column 参数";
          }
          const result = await client.getHover(uri, args.line, args.column);
          return formatHover(result);
        }

        case "diagnostics": {
          const result = await client.getDiagnostics(uri);
          return formatDiagnostics(result);
        }

        default:
          return \`未知操作: \${args.operation}\`;
      }
    } finally {
      await client.closeDocument(uri);
    }
  },
};

/* ──── 格式化函数 ──── */

function formatCompletion(result: unknown): string {
  // LSP CompletionList 格式
  const list = (result as any)?.items ?? (result as any) ?? [];
  if (!Array.isArray(list) || list.length === 0) return "没有补全建议。";

  return list
    .slice(0, 20)
    .map((item: any) => {
      const detail = item.detail ? \` — \${item.detail}\` : "";
      const kind = item.kind ? \`[\${["","Text","Method","Function","Constructor","Field","Variable","Class","Interface","Module","Property"][item.kind] || item.kind}]\` : "";
      return \`  \${kind} \${item.label}\${detail}\`;
    })
    .join("\n");
}

function formatLocations(result: unknown): string {
  if (!result) return "没有找到。";

  const locations = Array.isArray(result) ? result : [result];
  if (locations.length === 0) return "没有找到。";

  return locations
    .slice(0, 30)
    .map((loc: any) => {
      const uri = loc.uri || loc.targetUri || "?";
      const range = loc.range || loc.targetRange || {};
      const start = range.start || {};
      return \`  \${uri}:\${start.line + 1}:\${(start.character || 0) + 1}\`;
    })
    .join("\n");
}

function formatHover(result: unknown): string {
  if (!result) return "没有悬停信息。";

  const contents = (result as any)?.contents;
  if (!contents) return "没有文档。";

  const value = typeof contents === "string"
    ? contents
    : Array.isArray(contents)
      ? contents.map((c: any) => c.value || c).join("\\n")
      : contents.value || JSON.stringify(contents);

  return value;
}

function formatDiagnostics(result: unknown): string {
  const items = (result as any)?.items ?? [];
  if (!Array.isArray(items) || items.length === 0) return "没有诊断错误。";

  return items.map((d: any) => {
    const sev = ["", "错误", "警告", "信息", "提示"][d.severity] || "未知";
    const pos = d.range?.start
      ? \`:\${d.range.start.line + 1}:\${d.range.start.character + 1}\`
      : "";
    return \`[\${sev}]\${pos} \${d.message}\`;
  }).join("\n");
}`}</CodeBlock>
